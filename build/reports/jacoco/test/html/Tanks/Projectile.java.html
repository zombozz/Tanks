package Tanks;

import processing.core.PApplet;
import static processing.core.PApplet.cos;
import static processing.core.PApplet.sin;

public class Projectile {
    private PApplet parent;
    private float x;
    private float y;
    private float power;
    private float CELLSIZE;
    private float rotationAngle;
    private int ScreenY= App.HEIGHT;
    private int ScreenX = App.WIDTH;

    public Projectile(PApplet parent, float x, float y, float power, int CELLSIZE, float rotationAngle) {
        this.parent = parent;
        this.x = x;
        this.y = y;
        this.power = power;
        this.rotationAngle = rotationAngle - 90;
        this.CELLSIZE = CELLSIZE;
    }
    public void update() {       
        float angleInRadians = PApplet.radians(rotationAngle);
        float vx = power * cos(angleInRadians); // Calculate the x-component of velocity
        float vy = power * sin(angleInRadians); // Calculate the y-component of velocity

        // vy -= 3.6;
        x += vx; // Update x position
        y += vy; // Update y position
    }

    public void display() {
        parent.fill(255, 0, 0); // Set color to red
        parent.ellipse(x, y, CELLSIZE/8, CELLSIZE/8); // Draw the projectile as a circle
    }

    public boolean projectileLanded() {
        // return (y < 0 || y > ScreenY || x < 0 || x < ScreenX);
        return (false);
    }
}
